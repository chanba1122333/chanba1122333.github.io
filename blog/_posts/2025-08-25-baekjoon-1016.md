## 백준 1016번: 제곱ㄴㄴ수

[문제 링크](https://www.acmicpc.net/problem/1016)

코드

```cpp
# 백준 1016번: 제곱ㄴㄴ수

#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
#define FOR(i, a, A) for (ll i = a; i < A; ++i)

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll min, max;
    cin >> min >> max;
    ll range = max - min + 1;
    bool sqrssnum[range];
    FOR(i, 0, range)
    {
        sqrssnum[i] = false;
    }
    for (ll i = 2; i * i <= max; ++i)
    {
        ll square = i * i;
        ll wtm = (min + square - 1) / square;
        ll cs = wtm * square;
        for (ll num = cs; num <= max; num += square)
        {
            ll index = num - min;
            if (index >= 0 && index < range)
            {
                sqrssnum[index] = true;
            }
        }
    }

    ll sqrssnum_count = 0;
    FOR(i, 0, range)
    {
        if (!sqrssnum[i])
        {
            sqrssnum_count++;
        }
    }

    cout << sqrssnum_count << '\n';

    return 0;
}

```
