## 백준 1016번: 제곱ㄴㄴ수

[문제 링크](https://www.acmicpc.net/problem/1016)

 코드

```cpp
# 백준 1016번: 제곱ㄴㄴ수

#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
#define FOR(i, a, A) for (ll i = a; i < A; ++i)

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll min, max;
    cin >> min >> max;
    ll range = max - min + 1;
    bool sqrssnum[range];
    FOR(i, 0, range)
    {
        sqrssnum[i] = false;
    }
    for (ll i = 2; i * i <= max; ++i)
    {
        ll square = i * i;
        ll wtm = (min + square - 1) / square;
        ll cs = wtm * square;
        for (ll num = cs; num <= max; num += square)
        {
            ll index = num - min;
            if (index >= 0 && index < range)
            {
                sqrssnum[index] = true;
            }
        }
    }

    ll sqrssnum_count = 0;
    FOR(i, 0, range)
    {
        if (!sqrssnum[i])
        {
            sqrssnum_count++;
        }
    }

    cout << sqrssnum_count << '\n';

    return 0;
}
```

[문제 링크](https://www.acmicpc.net/problem/1436)

```cpp
# 백준 1436번: 영화감독 숌

using namespace std;
typedef long long ll;

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll n;
    cin >> n;
    ll count = 0;
    ll num = 666;
    while (1)
    {
        string s = to_string(num);
        if (s.find("666") != string::npos)
        {
            count++;
        }
        if (count == n)
        {
            cout << num << '\n';
            break;
        }

        num++;
    }
    return 0;
}
```

## 백준 18870번: 좌표 압축

[문제 링크](https://www.acmicpc.net/problem/18870)

코드

```cpp
# 백준 18870번: 좌표 압축
#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef vector<ll> vll;
typedef vector<vpll> vvpll;
#define FOR(i, a, A) for (ll i = a; i < A; ++i)

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll n;
    cin >> n;
    vll v(n);
    FOR(i, 0, n)
    {
        cin >> v[i];
    }
    vll sorted = v;
    sort(sorted.begin(), sorted.end());
    sorted.erase(unique(sorted.begin(), sorted.end()), sorted.end());

    FOR(i, 0, n)
    {
        ll idx = lower_bound(sorted.begin(), sorted.end(), v[i]) - sorted.begin();
        cout << idx << " ";
    }

    return 0;
}
```

## 백준 10815번: 숫자 카드

[문제 링크](https://www.acmicpc.net/problem/10815)

코드

```cpp
# 백준 10815번: 숫자 카드

#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
typedef vector<ll> vll;

#define FOR(i, a, A) for (ll i = a; i < A; ++i)

int main(void)
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    ll a, n, m, card1, card2;
    vll deck1, deck2;
    cin >> n;
    FOR(i, 0, n)
    {
        cin >> a;
        deck1.emplace_back(a);
    }
    cin >> m;
    FOR(i, 0, m)
    {
        cin >> a;
        deck2.emplace_back(a);
    }
    sort(deck1.begin(), deck1.end());
    for (ll card : deck2)
    {
        if (binary_search(deck1.begin(), deck1.end(), card))
        {
            cout << 1 << " ";
        }
        else
        {
            cout << 0 << " ";
        }
    }
    cout << '\n';
    return 0;
}
```
